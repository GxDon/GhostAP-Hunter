import subprocess
import time
from collections import defaultdict

known_networks = defaultdict(set)

def scan():
    result = subprocess.check_output(
        ["nmcli", "-t", "-f", "SSID,BSSID,SECURITY,SIGNAL", "dev", "wifi"]
    )
    lines = result.decode().strip().splitlines()

    networks = []

    for line in lines:
        if line.strip() == "":
            continue

        parts = line.split(":", 3)
        if len(parts) < 4:
            continue

        ssid, bssid, security, signal = parts
        networks.append({
            "ssid": ssid,
            "bssid": bssid,
            "security": security.upper(),
            "signal": signal
        })

    return networks

def analyze(networks):
    print("\n🧠 Analizando...\n")

    for net in networks:
        ssid = net["ssid"]
        bssid = net["bssid"]
        security = net["security"]

        # Guardar historial
        known_networks[ssid].add(bssid)

        # 🚨 Detectar Evil Twin
        if len(known_networks[ssid]) > 1:
            print(f"🚨 POSIBLE EVIL TWIN: {ssid}")
            print(f"   MACs detectadas: {known_networks[ssid]}")

        # ⚠️ Seguridad débil
        if security == "--":
            print(f"⚠️ RED ABIERTA: {ssid}")
        elif "WEP" in security:
            print(f"⚠️ WEP inseguro: {ssid}")
        elif "WPA1" in security:
            print(f"⚠️ WPA1 obsoleto: {ssid}")
        else:
            print(f"✅ Segura: {ssid}")

def monitor():
    print("📡 Iniciando monitoreo WiFi...\n")

    while True:
        networks = scan()
        analyze(networks)
        time.sleep(10)

# ===== RUN =====
monitor()
